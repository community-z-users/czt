\end{schema}

\zastate
\begin{schema}{AState}
   x: \power \nat
\where
  x \neq \emptyset
\end{schema}

\zastinit
\begin{schema}{ASInit}
   AState~'
\where
   x' = \{ 0 \}
\end{schema}

\zcstate
\begin{schema}{CState}
  y: \seq \nat
\where
   y \neq \langle \rangle
\end{schema}

\zcstinit
\begin{schema}{CStInit}
   CState~'
\where
  y' = \langle 0 \rangle
\end{schema}
