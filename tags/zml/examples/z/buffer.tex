\documentclass{article}
\usepackage{z-eves}
\begin{document}

\begin{zed}
  [MSG]
\end{zed}

\begin{axdef}
  MaxSize: \nat
\end{axdef}

\begin{schema}{Buffer}
  items: \seq  MSG
\where
  \# items \leq  MaxSize
\end{schema}

\begin{schema}{BufferINIT}
  Buffer
\where
  items = \langle \rangle
\end{schema}

%\begin{theorem}{InitBufferPossible}
%\exists  Buffer @ BufferINIT
%\end{theorem}

\begin{schema}{Join}
  \Delta Buffer\\
  msg?: MSG
\where
  \# items < MaxSize\\
  items' = items \cat  \langle msg?\rangle
\end{schema}

%\begin{theorem}{JoinPrecondition}
%\forall  Buffer; msg?: MSG | \# items < MaxSize @ \pre  Join
%\end{theorem}

\begin{schema}{Leave}
  \Delta Buffer\\
  msg!: MSG
\where
  \# items > 0\\
  items = \langle msg!\rangle  \cat  items'
\end{schema}

%\begin{theorem}{LeavePrecondition}
%\forall  Buffer | \# items > 0 @ \pre  Leave
%\end{theorem}

%\begin{theorem}{FIFO}
%\forall  Buffer; msg: MSG @ (Join[msg? := msg] \implies  \# items' = \# items + 1 \land  last items' = msg) \land  (Leave \implies  \# items' = \# items - 1 \land  msg! = head items)
%\end{theorem}

\begin{schema}{Flush}
  \Delta Buffer\\
  out!: \seq  MSG
\where
  out! = items\\
  items' = \langle \rangle
\end{schema}

\begin{schema}{Count}
  \Xi Buffer\\
  m?: MSG\\
  count!: \nat
\where
  count! = \# (items \rres  \{m?\})
\end{schema}

\begin{zed}
NewJoin == [\Xi Buffer; m?: MSG | m? \notin  \ran  items] \land  Join
\end{zed}

\begin{schema}{Duplicate}
  \Delta Buffer
\where
  \# items \in  1 .. MaxSize - 1\\
  items' = items \cat  \langle head~items\rangle
\end{schema}

\begin{schema}{Size}
  \Xi Buffer\\
  n!: \nat
\where
  n! = \# items
\end{schema}

\begin{schema}{Titanic}
  \Delta Buffer\\
  msgs?: \seq  MSG
\where
  items' = (1 .. MaxSize) \dres  (items \cat  msgs?)
\end{schema}

\begin{schema}{Penguin}
  \Delta Buffer\\
  msgs?: \seq  MSG
\where
  \exists  s: \seq  MSG @ s \cat  items' = items \cat  msgs? \land  (s \neq  \langle \rangle  \implies  \# items' = MaxSize)
\end{schema}

\end{document}
