From Z standard, section 8.3:

  "Different associativities shall not be used at the same precedence level
  by operator template paragraphs in the same scope."

\begin{zsection}
\SECTION sect1
\end{zsection}
%%Zinword \e e
\begin{zed}
  \function 50 \leftassoc (\_ \e \_)
\end{zed}

\begin{zsection}
\SECTION sect2
\end{zsection}

%%Zinword \f f
\begin{zed}
  \function 50 \rightassoc (\_ \f \_)
\end{zed}

\begin{zsection}
\SECTION sect3 \parents sect1, sect2
\end{zsection}
