\begin{axdef}
  \delta : \atime
\end{axdef}

Here, we specify a $Clock$ class, which provides active objects for
keeping the time.

\begin{class}{Clock}
\project (time)\\
\begin{state}
  time : \atime
\end{state}\\
\begin{init}
  time = 0
\end{init}\\
\begin{op}{Tick}
  \Delta (time)
\where
  \delta < 1\\
  time' = time + 1
\end{op}\\
\Main \sdef \mu C @ (Tick @ \WaitUntil~1) %\semi C
\end{class}
