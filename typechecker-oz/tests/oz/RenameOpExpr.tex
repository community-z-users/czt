Check renames by conjoining with incompatible types.
Check non-renames by hiding the names.

\begin{class}{A}
  Op1 \sdef [| true] [a/x]\\
  Op2 \sdef [x : \nat] [a/x]\\
  Op3 \sdef [x : \nat] [x/a]\\
  Test2a \sdef Op2 \hide (a)\\
  Test2b \sdef Op2 \land [x : \power\nat]\\
  Test3 \sdef Op3 \hide (x)\\
\end{class}

  Op3 \sdef [x : \nat; y : \nat] [a/x, b/y]\\
  Op4 \sdef [x : \nat; y : \nat] [a/x, a/y]\\
  Test1a \sdef Op1 \land [x : \power \nat]\\
  Test1b \sdef Op1 \hide (y,a)\\
  Test2a \sdef Op2 \land [y : \power \nat]\\
  Test2b \sdef Op2 \hide (x,a)\\
  Test3a \sdef Op3 \land [x, y : \power \nat]\\
  Test3b \sdef Op3 \hide (a,b)\\
  Test4a \sdef Op4 \land [x,y : \power \nat]\\
