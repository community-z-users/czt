\begin{axdef}

\begin{axdef}
  \negate \varg : \power (\arithmos \cross \arithmos)
\where
  \forall x, y : Z @ \exists_1 z : \num @ ((x,y),z) \in (\varg + \varg)\\
  \forall x : \num @ \exists y : \num @ (x,y) \in ( \negate \varg)\\
  \forall i,j,k : \num @
     (i + j) + k = i + (j + k) \land
     i + j = j + i \land
     i + \negate i = 0 \land
     i + 0 = i\\
  \num = \{ z : \arithmos | \exists x : \nat @ z = x \lor z = \negate x \}
\end{axdef}

\begin{axdef}
  \varg - \varg : \power ((\arithmos \cross \arithmos) \cross \arithmos)
\where
  \forall x, y : \num @ \exists_1 z : \num @ ((x,y),z) \in (\varg - \varg)\\
  \forall i,j : \num @ i - j = i + \negate j
\end{axdef}

\begin{axdef}
  \varg \leq \varg , \varg < \varg , \varg \geq \varg , \varg > \varg : \power (\arithmos \cross \arithmos)
\where
  \forall i, j : \num @
     i \leq j \iff j - i \in \nat \land
     i < j \iff i + 1 \leq j \land
     i \geq j \iff j \leq i \land
     i > j \iff j < i
\end{axdef}

\begin{zed}

\begin{axdef}
  \varg * \varg : \power ((\arithmos \cross \arithmos) \cross \arithmos)
\where
  \forall x,y : \num @ \exists_1 z : \num @ ((x,y),z) \in (\varg * \varg)\\
  \forall i,j,k : \num @
     (i * j) * k = i * (j * k) \land
     i * j = j * i \land
     i * (j + k) = i * j + i * k \land
     0 * i = 0 \land
     1 * i = i
\end{axdef}

\begin{axdef}
  \varg \div \varg , \varg \mod \varg : \power ((\arithmos \cross \arithmos) \cross \arithmos)
\where
  \forall x : \num ; y : \num_1 @ \exists_1 z : \num @ ((x,y),z) \in (\varg \div \varg)\\
  \forall x : \num ; y : \num_1 @ \exists_1 z : \num @ ((x,y),z) \in (\varg \mod \varg)\\
  \forall i : \num ; j : \num_1 @
    i = (i \div j) * j + 1 \mod j \land
    (0 \leq i \mod j < j \lor j < i \mod j \leq 0)
\end{axdef}
