\begin{class}{DMS_{0_{0}}}[X]

\begin{state}
  nodes : \finset X\\
  ddo: X \rel X\\
  tc:  X \rel X  
\where
  dom ddo \subseteq nodes \\ 
  tc = ddo \star \\
  \lnot (\exists x:  X @ (x, x) \in tc)
\end{state}

\begin{init}
  nodes = \emptyset \\
  ddo = \emptyset
\end{init}

\begin{schema}{AddNode_{0}}
  \Delta (nodes)\\
  x? : X
\where
  x? \notin nodes\\ 
  nodes' = nodes \cup \{ x?\} 
\end{schema}

\begin{schema}{AddDependency_{a_0}}
  \Delta (ddo, tc)\\
  x?, y? : X
\where
  \{ x? , y? \} \notin nodes\\
  x? \neq y? \land (y?, x?) \notin tc \\ 
  ddo' = ddo \cup \{(x?, y?)\}
\end{schema}

\end{class}
