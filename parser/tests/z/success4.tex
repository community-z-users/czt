z : 1 \upto 10

1 \leq z \leq 10 \\

\end{schema}

\begin{genschema}{Pool}{RESOURCE}
owner: RESOURCE \pfun USER \\
free : \power RESOURCE
\where
(\dom owner) \uni free = RESOURCE \\
(\dom owner) \int free = \emptyset
