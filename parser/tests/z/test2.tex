\begin{axdef}
up, down: SQUARE \fun DIAGONAL\\
\where \forall f : FILE; r : RANK @ up(f, r) = r - f \land down(f, r)
= r + f
\end{axdef}

\begin{schema}{Queens}
squares : FILE \bij RANK \\
\where \{squares \dres up, squares \dres down \} \subseteq SQUARE
\tinj DIAGONAL
\end{schema}
