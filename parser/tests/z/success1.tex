\begin{zed}
[NAME, DATE]
\end{zed}

\begin{schema}{BirthdayBook}
known : \power NAME \\
birthday : NAME \pfun DATE
\where 
known = \dom birthday
\end{schema}

\begin{schema}{AddBirthday}
\Delta BirthdayBook \\
name? : NAME \\
date? : DATE
\where
name? \notin known \\
birthday' = birthday \cup \{ name? \mapsto date? \}
\end{schema}

\begin{schema}{FindBirthday}
\Xi Birthday \\
name? : NAME \\
date! : DATE
\where 
name? \mem known \\
date! = birthday(name?)
\end{schema}

\begin{schema}{Remind}
\Xi BirthdayBook \\
today? : DATE \\
cards! : \power NAME
\where
cards! = \{ n: known | birthday (n)  = today? \}
\end{schema}

\begin{schema}{InitBirthdayBook}
BirthdayBook 
\where
known = \emptyset
\end{schema}
