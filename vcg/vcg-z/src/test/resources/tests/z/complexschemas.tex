State schema with generics
\begin{schema}{State}
   x: \nat \\
   y: \power~\nat \\
   f: \nat \pfun \nat 
\where
   y \cross y \subseteq f \\
   x \in y \\
   y \cap y = y
\end{schema}

\begin{schema}{Op}
   \Delta State\\
   in?: \nat \\
   out!: \nat 
\where
   in? \neq out!
\end{schema}

\end{zed}
