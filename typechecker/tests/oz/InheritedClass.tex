  \end{state}\\

  \end{op}

\begin{class}{B}
  A\\
  l1 == 2\\
  \begin{state}
    n : \nat\\
    pn : \power \nat
  \end{state}\\
  \begin{op}{Op2}
    x? : \power \nat
  \end{op}
\end{class}

\begin{class}{C}
  \begin{axdef}
    a : \poly A\\
    b : B
  \where
    a.n = 1\\
    b.pn = \{1\}\\
    b.pn = \{a.n\}
  \end{axdef}\\
  \begin{state}
    sa : \poly A\\
    sb : B
  \end{state}\\
  Op3 \sdef sa.Op1\\
  Op4 \sdef sb.Op1\\
  Op5 \sdef sb.Op2
\end{class}

Multiple inheritance
\begin{class}{D}
  B\\
  C\\
  \begin{state}
    d : \nat
  \where
    d = n\\
    d \in pn\\
    a \in \poly A\\
    b \in B
  \end{state}\\
  Op4 \sdef Op3 \land sa.Op1\\
  Op6 \sdef Op3 \land sa.Op1\\
  Op7 \sdef Op4 \land sb.Op1
\end{class}

Test that overridden operations have their signatures conjoined with
their parent.
\begin{class}{E}
  A\\
  \begin{op}{Op1}
    y? : \power\nat
  \where
    %test that variables from the super definition 
    %are implicitly declared
    x? \in y?  
  \end{op}\\
  Test1a \sdef Op1 \hide (x?,y?)
\end{class}

\begin{class}{F}
  E\\
  \begin{op}{Op1}
    x? : \nat\\
    y? : \power \nat\\
    z? : \power\power \nat
  \end{op}\\
  Test1b \sdef Op1 \hide (x?,y?,z?)\\
  Test1c \sdef Op1 \land [x? : \nat; y? : \power\nat; z? : \power\power\nat]
\end{class}

Test that the superclasses signature is not altered by the subclass
\begin{class}{G}
  \begin{state}
    e : E
  \where
    e.self = e
  \end{state}\\
  Op1 \sdef e.Op1 \land [z? : \nat]
\end{class}
