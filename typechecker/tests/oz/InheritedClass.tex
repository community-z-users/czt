\begin{class}{A}
  \begin{state}
    n : \nat
  \end{state}\\
  \begin{op}{Op1}
    x? : \nat
  \end{op}
\end{class}

\begin{class}{B}
  A\\
  \begin{state}
    pn : \power \nat
  \end{state}\\
  \begin{op}{Op2}
    x? : \power \nat
  \end{op}
\end{class}

\begin{class}{C}
  \begin{axdef}
    a : \poly A\\
    b : B
  \where
    a.n = 1\\
    b.pn = \{1\}\\
    b.pn = \{a.n\}
  \end{axdef}\\
  \begin{state}
    sa : \poly A\\
    sb : B
  \end{state}\\
  Op3 \sdef sa.Op1\\
  Op4 \sdef sb.Op1\\
  Op5 \sdef sb.Op2
\end{class}

Multiple inheritance
\begin{class}{D}
  B, C\\
  \begin{state}

  \where
    d = n\\
    d \in pn\\
    a \in \poly A\\
    b \in B
  \end{state}\\
  Op4 \sdef Op3 \land sa.Op1\\
  Op6 \sdef Op3 \land sa.Op1\\
  Op7 \sdef Op4 \land sb.Op1
\end{class}
